	public string getBarcode(string productID)
        {
            SqlConnection con = DatabaseHelper.getConnection();
            string barcode = "";
            try
            {
                con.Open();
                SqlCommand cmd = new SqlCommand($"select Barcode from Product where ProductId = '{productID}'", con);
                // Khai báo DataTable để lưu kết quả từ câu truy vấn
                DataTable dt = new DataTable();
                SqlDataAdapter adp = new SqlDataAdapter(cmd);
                adp.Fill(dt);
                // Kiểm tra xem có bất kỳ dòng dữ liệu nào được trả về hay không
                if (dt.Rows.Count > 0)
                {
                    // Nếu có dòng dữ liệu được trả về, lấy giá trị barcode từ cột đầu tiên của dòng đó
                    barcode = dt.Rows[0]["Barcode"].ToString();
                }
            }
            catch (SqlException ex)
            {
                // In lỗi ra console để kiểm tra và gỡ rối
                Console.WriteLine($"SQL Exception: {ex.Message}");
                return null;
            }
            finally
            {
                con.Close();
            }
            return barcode;
        }
